## 快速命令

```bash
# 查看搜索进度
cd code && python scripts/update_search_progress.py

# 查看最优模型配置
cat newest_models/best_model_config.json

# 训练最优配置的模型
python run.py --is_training 1 --model VoltageTimesNet_v2 --data RuralVoltage \
  --root_path ./dataset/RuralVoltage/realistic_v2/ \
  --enc_in 16 --c_out 16 --d_model 128 --e_layers 3 --d_ff 256 \
  --top_k 2 --num_kernels 8 --learning_rate 0.0042 --batch_size 32 \
  --seq_len 50 --anomaly_ratio 2.08 --dropout 0.017
```
